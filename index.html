My solution to a Codewars problem:


function strCount(obj) {
  switch (typeof obj) {
    case "string": return 1;
    case "object": return Object.values(obj)
      .filter(x => x)
      .map(strCount)
      .reduce((a, b) => a + b, 0);
    default: return 0;
  }
}
